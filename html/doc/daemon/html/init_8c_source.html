470 {
471  size_t size = sizeof(disapperingWidgetsArr) / sizeof(disapperingWidgetsArr[0]);
472  int i;
473  for(i = 0; i < size; ++i)
474  {
475  if(disapperingWidgetsArr[i] == NULL)
476  {
477  fprintf(stderr, "ERROR: current pointer number %d to a disappearing widget is NULL\n", i);
478  return;

482  case HIDE:

484  break;
485  case SHOW:
486  gtk_widget_show(disapperingWidgetsArr[i]);

488  default:
489  fprintf(stderr, "ERROR: Unknown state of visibility\n");

491  }
492  }
