158 const string CommandsDispatcher::execCommand(const string& command) const
159 {
160  if(command.empty())
161  {
162  writeToLog(string("ERROR: execCommand(): The given command is empty\n").c_str(), TAG);
163  return ERR;
164  }
165 
166  list<string> commandList = parseCommandStr(command); // list: {command_name, parameter, parameter, ...}
167  if(commandList.empty())
168  {
169  writeToLog(string("ERROR: execCommand(): Can't parse the given command '" + command + "'\n").c_str(), TAG);
170  return ERR;
171  }
172 
173  auto commandIt = commandList.begin();
174  string name = *commandIt;
175 
176  map<string, Command*>::const_iterator it = commands_.find(name);
177  if(it == commands_.end())
178  {
179  writeToLog(string("ERROR: execCommand(): The command '" + command + "' doesn't exist\n").c_str(), TAG);
180  return ERR;
181  }
182 
183  if(commandList.size() == 1) // command without params
184  return (*it).second->execute();
185 
186  commandList.pop_front(); // drop the command's name
187  return (*it).second->execute(commandList);
188 }
