218  @Override
219  public void processActivityResult(final int requestCode, final Intent data) {
220  if (requestCode == UI.REQUEST_CODE_EDIT) {
221  final int action_type = data.getIntExtra(EditDesktopDataActivity.ACTION_TYPE, EditDesktopDataActivity.ACTION_ERR);
222  switch (action_type) {
223  case EditDesktopDataActivity.ACTION_EDIT:
224  chgCurrentlyEditedDesktopData(data.getStringExtra(SINGLE_DESKTOP_DATA), data.getBooleanExtra(IS_DEFAULT, false));
225  host_connection_checked = false;
226  break;
227  case EditDesktopDataActivity.ACTION_DEL:
228  removeCurrentlyEditedDesktopData();
229  saveDesktopsDatas();
230  host_connection_checked = false;
231  break;
232  case EditDesktopDataActivity.ACTION_CANCEL:
233  host_connection_checked = false;
234  redraw = true;
235  curEditedDesktopNameInd = NO_EDIT;
236  break;
237  case EditDesktopDataActivity.ACTION_ERR:
238  Log.e(TAG, "There is some error at transferring an action type from the activity '" + EditDesktopDataActivity.class.getName() + "'");
239  host_connection_checked = true;
240  break;
241  default:
242  Log.e(TAG, "Unknown action type " + action_type + " has been received from the activity '" + EditDesktopDataActivity.class.getName() + "'");
243  return;
244  }
245  }
246 
247  if (requestCode == REQUEST_CODE_CONNECTING) {
248  host_connection_checked = true;
